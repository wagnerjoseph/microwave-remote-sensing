def preprocess_data(dc, nodata_val=-9999, scale_factor=0.1, band='VV'):
    '''
    This function preprocesses the Sentinel-1 data by handling the nodata values and scaling the data.
    Also it only loads the VV band.
    '''
    return (dc.where(dc != nodata_val)*scale_factor)[band].median(dim='time')

def load_data(aoi:list[tuple], timeframe:list[str], spatial_ext:float=0.125, epsg_code:int=4326, spatial_res:float=0.0002) -> list:
    '''
    This function helps to load Sentinel-1 data from the EODC STAC API.

    Params:
    -------
    - aoi (list[tuple]): List of AOIs as center points (lat, lon)
    - timeframe (list[str]): List of timeframes in the format 'YYYY-MM-DD/YYYY-MM-DD'
    - spatial_ext (float): Spatial extent around the center point
    - epsg_code (int): EPSG code of the coordinate reference system
    - spatial_res (float): Spatial resolution of the data

    Returns:
    -------
    - list: List of xarray datasets
    '''
    epsg = CRS.from_epsg(epsg_code) 
    dx = spatial_res 
    datasets = []
    for area in aoi:
        for time in timeframe:
            # Set the spatial extent
            latmin, latmax = area[0] - spatial_ext, area[0] + spatial_ext
            lonmin, lonmax = area[1] - spatial_ext, area[1] + spatial_ext
            bounds = (lonmin, latmin, lonmax, latmax)

            # Search items
            items = pystac_client.Client.open("https://stac.eodc.eu/api/v1").search(
                # intersects=geom,
                bbox=bounds,
                collections=["SENTINEL1_SIG0_20M"],
                datetime=time,
                limit=100,
            ).item_collection()

            # define a geobox for my region
            geobox = GeoBox.from_bbox(bounds, crs=epsg, resolution=dx)

            # lazily combine items
            sig0_dc = odc.stac.stac_load(
                items,
                #bbox=bounds,
                bands=["VV", "VH"],
                chunks={'time': 5, 'x': 600, 'y': 600},
                epsg=epsg,
                geobox=geobox,
                resampling="bilinear",
            )
            datasets.append(sig0_dc)
    return datasets

def calc_mean(array, convert:bool= False) -> dict:
    '''
    This function calculates the mean of the Sentinel-1 data.
    '''
    if convert:
        array = db2lin(array) # convert to linear scale
        mean_lin = array.mean().compute() # compute mean
        mean = lin2db(mean_lin)# convert back to dB
    else:
        mean = array.mean().compute()

    return float(mean)
